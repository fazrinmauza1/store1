```bash
> pkg update && pkg upgrade
> pkg install git -y
> pkg install nodejs -y
> pkg install ffmpeg -y
> pkg install imagemagick -y
> git clone https://github.com/dvlgaming/devilbotzv5
> cd devilbotzv5
> npm install
```
###### Run
```bash
> npm start/node main.js
```
